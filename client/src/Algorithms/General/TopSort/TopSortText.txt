const nodes = [0,1,2,3];
const edgeList = [[{to: 1}, {to: 2}], [{to: 2}, {to: 3}], [], [{to: 2}]];

function dfs(node) {
    if (visited[node]) {
        return;
    }

    visited[node] = true;

    for (let i = 0; i < edgeList[node].length; i++) {
        dfs(edgeList[node][i].to);
    }

    nodesTopSorted.unshift(node);
}

const nodesTopSorted = [];
const visited = [];

for (let i = 0; i < nodes.length; i++) {
    if (!visited[i]) dfs(i);
}
